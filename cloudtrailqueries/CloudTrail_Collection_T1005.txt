SELECT *
FROM logs
WHERE event_source = 's3.amazonaws.com'
  AND event_name   = 'GetObject'
  AND remote_ip NOT LIKE '10.%'
  AND remote_ip NOT LIKE '192.168.%'
  AND remote_ip NOT LIKE '127.%'
  AND remote_ip NOT LIKE '172.31.%'
  AND remote_ip NOT LIKE '158.62.32.52'
  AND bucket     IS NOT NULL
  AND object_key IS NOT NULL
  AND object_key NOT LIKE '%favicon.ico';
