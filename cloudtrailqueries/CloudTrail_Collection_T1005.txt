SELECT *
FROM logs
WHERE event_source    = 's3.amazonaws.com'
  AND event_name      = 'GetObject'
  AND sourceIPAddress NOT LIKE '10.%'
  AND sourceIPAddress NOT LIKE '192.168.%'
  AND sourceIPAddress NOT LIKE '127.%'
  AND sourceIPAddress NOT LIKE '172.31.%'
  AND sourceIPAddress NOT LIKE '158.62.32.52'
  AND requestParameters.bucketName IS NOT NULL
  AND requestParameters.key        IS NOT NULL
  AND requestParameters.key NOT LIKE '%favicon.ico';
